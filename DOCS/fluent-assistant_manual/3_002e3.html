3.3 3.3

(define fibonacci-number-tail-recursive
  (lambda (n)
    (define aux
      (lambda (acc1 acc2 current-n)
	(if (equal? current-n n)
	    (+ acc1 acc2)
	    (aux acc2
		 (+ acc1 acc2)
		 (add1 current-n)))))
    (cond
     ((equal? n 1) 0)
     ((equal? n 2) 1)
     (else (aux 0 1 3)))))
